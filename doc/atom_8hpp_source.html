289  inline void append (const AtomEvent& ev)
290  {
291  if (AtomEvent* pos = lv2_atom_sequence_end (&p_seq->body, p_seq->atom.size))
292  {
293  memcpy (pos, &ev, sizeof (AtomEvent));
294  memcpy (pos + 1, LV2_ATOM_BODY_CONST (&ev.body), ev.body.size);
295  ((LV2_Atom*) p_seq)->size += sizeof (LV2_Atom_Event) + lv2_atom_pad_size (ev.body.size);
296  }
297  }
298 
301  inline void insert (const AtomEvent& ev)
302  {
303  const uint32_t evsize = sizeof (LV2_Atom_Event) + lv2_atom_pad_size (ev.body.size);
304  AtomEvent* pos = lv2_atom_sequence_end (&p_seq->body, p_seq->atom.size);
305  LV2_ATOM_SEQUENCE_FOREACH (p_seq, iter)
306  {
307  if (iter->time.frames > ev.time.frames)
308  {
309  memmove ((uint8_t*)iter + evsize, iter,
310  (uint8_t*)pos - (uint8_t*)iter);
311  pos = iter;
312  break;
313  }
314  }
315 
316  if (pos)
317  {
318  memcpy (pos, &ev, sizeof (AtomEvent));
319  memcpy (pos + 1, LV2_ATOM_BODY_CONST (&ev.body), ev.body.size);
320  ((LV2_Atom*) p_seq)->size += evsize;
321  }
322  }
